1. Określanie właściwości

auto.marka = "Opel";
auto["cena"] = 25000;

2. Odwołanie do pól i metod

console.log(auto.kolor);
auto.jedz();

3. Cechy konstruktora

  • uruchamia się automatycznie
  • ma nazwę klasy
  • nie zwraca wartości
  • może mieć argumenty
  • używa się z new

4. Przykład obiektu

var osoba = {
 nazwisko: "Kowalski",
 imie: "Jan",
 zawod: "piekarz",
 wyswietl: function() {
  document.write(this.nazwisko + " " + this.imie);
 }
}

5. This

this odnosi się do aktualnego obiektu. Pola i metody oddzielamy przecinkami.

6. Konstruktor

function Klient(nazwisko, imie) {
 this.nazwisko = nazwisko;
 this.imie = imie;
}

7. New

var osoba1 = new Klient("Kowalski", "Jan");

8. Prototype

Prototype pozwala dodawać wspólne metody dla wszystkich obiektów.

function Klient(){}

Klient.prototype.wypisz = function() {
 document.write("Test");
